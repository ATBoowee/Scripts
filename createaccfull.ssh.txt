#! /bin/bash

echo "Enter name of new backend account"
read backname

echo "Name you entered is $backname"


echo "Enter in the expiration date of account (YYYY-MM-DD)"
read expire

echo "Expiration date you entered is $expire"

temppass=$(openssl rand -base64 12)

useradd $backname

chage -E $expire $backname

echo -e "$temppass\n$temppass" | passwd $backname

chage -d 0 $backname

chage -l $backname

echo "Your temp password for user:$backname is $temppass"
